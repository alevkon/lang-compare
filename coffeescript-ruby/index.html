  puts '"if value":'
  values.each { |k, v| puts "#{k} - #{v ? 'true' : 'false'}" }
  puts ''
end

def nil_value(values)
  puts '"if value.nil?":'
  values.each { |k, v| puts "#{k} - #{v.nil? ? 'true' : 'false'}" }
  puts ''
end

def empty_value(values)
  puts '"if value.empty?":'
  values.each do |k, v|

  end

class ProccessElements
  def self.element(el_name)
    define_method "#{el_name}_element" do |content|
      "<#{el_name}>#{content}</#{el_name}>"
    end
  end
end

  element :div
  element :span
end
